State monad

The reason monads are so important for functional programming

type State state a = state -> (state, a)
instance Monad State state where
    return x = \s -> (s, x)
    bind :: (a -> State state b)       -> a -> State state b 
    --      (a -> state -> (state, b)) -> a -> (state -> (state, b))
    bind f x = let nextStep :: state -> (state, a)
                   nextStep oldState = f x oldState
               in  nextStep
                        

writeValue :: state -> State state ()
writeValue st = \_ -> (st, ())

readValue :: State state state
readValue = \st -> (st, st)

Avoiding reinventing the wheel


type Name = [String]
type NamedDataCollection a = Map Name a

type DataExtractor a b = a -> NamedDataCollection b
                    

instance Monad NamedDataCollection where
    return a = singleton [] a
    bind f x = let extractPairs :: (Name, a) -> [(Name, a)]
                   extractPairs (name, a) = 
                        toList (mapKeys (\k -> name ++ k) (f a))
               in fromList (bind extractPairs (toList x))
                    

customerOrderExtractor :: DataExtractor Customer Order
orderReportExtractor :: DataExtractor Order Report
customerReportExtractor :: DataExtractor Customer Report
customerReportExtractor = customerOrderExtractor >>> orderReportExtractor
                    

instance MonadPlus NamedDataCollection a
    mzero = empty
    mplus = union
