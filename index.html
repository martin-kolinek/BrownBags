Reactive Extensions

A library to compose asynchronous and event-based programs using observable collections and LINQ-style query operators.

Core interfaces


public interface IObservable<out T>
{
    IDisposable Subscribe(IObserver<T> observer);
}

public interface IObserver<in T>
{
    void OnCompleted();
    void OnError(Exception error);
    void OnNext(T value);
}

Basically observer pattern

We already have the observer pattern in C#


class Class
{
    public event EventHandler SomeEvent;
    
    private static void Method()
    {
        SomeEvent += (s, ev) => Console.WriteLine("Hello");
    }
}

So why would we introduce a new library for it?

OnCompleted / OnError

These methods allow us to complete a sequence of events 


class InformationProvider
{
    public event EventHandler Information;
    public void DoStuff();
}
                    

class InformationAggregator
{
    private List<InformationProvider> providers = 
        new List<InformationProvider>();
    public void AddProvider(InformationProvider provider)
    {
        providers.Add(provider);
        provider.Information += (s, ev) => Console.WriteLine("Info");
    }
    public void RemoveProvider(InformationProvider provider)
    { providers.Remove(provider); }
    public void DoStuff()
    {
        foreach(var provider in providers) { provider.DoStuff(); }
    }
}

What's wrong with this?

With IObservable


class InformationProvider
{
    public IObservable<Unit> Information;
    public void DoStuff();
}
                    

class InformationAggregator
{
    private List<InformationProvider> providers = 
        new List<InformationProvider>();
    public void AddProvider(InformationProvider provider)
    {
        providers.Add(provider);
        provider.Information.Subscribe(x => Console.WriteLine("Info"));
    }
    public void RemoveProvider(InformationProvider provider)
    { providers.Remove(provider); }
    public void DoStuff()
    {
        foreach(var provider in providers) { provider.DoStuff(); }
    }
}

Or even better


class InformationAggregator
{
    private List<InformationProvider> providers = 
        new List<InformationProvider>();
    public void AddProvider(InformationProvider provider) {
        providers.Add(provider);
        provider.Information.Subscribe(
            x => Console.WriteLine("Info"),
            exception => providers.Remove(provider),
            () => providers.Remove(provider));
    }
    public void DoStuff() {
        foreach(var provider in providers) { provider.DoStuff(); }
    }
}
Limiting lifetime of events can ease avoiding memory leaks.

Delegating events


class C1
{
    public event EventHandler<int> SomeEvent;
}
                    

class C2
{
    public C2(C1 c1)
    {
        c1.SomeEvent += (s, ev) =>
            {
                if(ev % 2 == 0 && SomeOtherEvent != null)
                {
                    SomeOtherEvent(s, ev);
                }
            }
    }
    
    public event EventHandler<int> SomeOtherEvent;
}

A better way?


class C2 {
    private C1 c1;
    public C2(C1 c1)
    {
        this.c1 = c1;
    }
    
    public event EventHandler<int> SomeOtherEvent
    {
        add
        {
            c1.SomeEvent += (s, ev) => { /*...*/ }
        }
        remove { /* ??? */ }
    }
}

With IObservable


class C1
{
    public IObservable<int> SomeObservable;
}
                    

class C2
{
    public IObservable<int> SomeOtherObservable { get; set; }
    public C2(C1 c1)
    {
        SomeOtherObservable = c1.SomeObservable.Where(p => p % 2 == 0);
    }
}

Event is just a delegate which can be invoked only from its class

By separating interfaces for invoking and subscribing, we achieve easier usage

Reactive extensions contain a lot of methods for combining, creating and transforming IObservables.
