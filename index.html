Replaying of events more or less for free


IObservable<MeasurementPoint> StartAnalysis(long analysisId)
{
  var analysisObservable = this.notificationService
    .DataObservable.Where(p => p.AnalysisId == analysisId);
  var beforeEnd = analysisObservable.TakeWhile(p => !p.IsLast);
  var end = analysisObservable.FirstAsync(p => p.IsLast);
  var result = beforeEnd.Merge(end).Replay();
  result.Connect();
  this.executionService.StartAnalysis(analysisId);
  return result;
}
